# model parameters
intercept <- 350
slope = 0
simulations <- 20000
sample_size <- 100

# empty dataframe to fill in the loop
df <- data.frame("p" = numeric(),
                 "r" = numeric(),
                 "cohen_d" = numeric())

# simulate power
for (i in 1:simulations) {
  x <- rep(c(0, 1), each = sample_size/2)
  y <- intercept + slope*x + rnorm(sample_size, mean = 10, sd = 5)
  sim_df <- data.frame(x, y)
  model <-  lm(y ~ x, data = sim_df)
  p_value <- summary(model)$coefficients[2,4]
  r <- sqrt(summary(model)$r.squared)
  cohen_d <- abs((summary(model)$coefficients[2, 1])/(sqrt(100)*summary(model)$coefficients[2, 2]))
  temp <- data.frame("p" = p_value,
                     "r" = r,
                     "cohen_d" = cohen_d)
  df <- rbind(temp, df)
}
# calculate power from simulation
type_I_error_rate <- (length(df[df$p <= 0.05, 1]))/simulations * 100
print(paste0("Type-I Error Rate: ", round(type_I_error_rate, 1)))

# model parameters
intercept <- 350
slope = 0
simulations <- 10000
sample_size <- 1000

# empty dataframe to fill in the loop
df <- data.frame("p" = numeric(),
                 "r" = numeric(),
                 "cohen_d" = numeric())
df_eyeball <- data.frame("p" = numeric(),
                         "r" = numeric(),
                         "cohen_d" = numeric())

# simulate power
for (i in 1:simulations) {
  x <- rep(c(0, 1), each = sample_size/2)
  y <- intercept + slope*x + rnorm(sample_size, mean = 50, sd = 15)
  sim_df <- data.frame(x, y)
  model <-  lm(y ~ x, data = sim_df)
  p_value <- summary(model)$coefficients[2, 4]
  r <- sqrt(summary(model)$r.squared)
  cohen_d <- abs((summary(model)$coefficients[2, 1])/(sqrt(100)*summary(model)$coefficients[2, 2]))
  temp <- data.frame("p" = p_value,
                     "r" = r,
                     "cohen_d" = cohen_d)
  df <- rbind(temp, df)
  if (cohen_d >= 0.18) {
    temp_eyeball <- data.frame("p" = p_value,
                               "r" = r,
                               "cohen_d" = cohen_d)
    df_eyeball <- rbind(temp_eyeball, df_eyeball)
  }
}
mean(df$cohen_d)
type_I_error_rate <- (length(df[df$p <= 0.05, 1]))/simulations * 100
